newsfeed_relation_if_not_acquaintance = {
    type = character

    text = {
        fallback = yes
        localization_key = "[Select_CString(Or(EqualTo_string(Scope.Char.Custom2('RelationToMe', second.Self), Localize('relation_acquaintance_female')), EqualTo_string(Scope.Char.Custom2('RelationToMe', second.Self), Localize('relation_acquaintance_male'))), '', Concatenate(Concatenate(Concatenate(Localize('CHARACTER_NAME_MY'), ' '), Scope.Char.Custom2('RelationToMe', second.Self)), ' '))]"
    }
}
