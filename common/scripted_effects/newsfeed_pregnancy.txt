newsfeed_notify_pregnancy_effect = {
    send_interface_message = {
        type = newsfeed_pregnancy
        title = {
            first_valid = {
                triggered_desc = {
                    trigger = {
                        OR = {
                            NOT = { exists = scope:father }
                            NOT = { newsfeed_is_interested_in_birth = { PERSON = scope:father } }
                        }
                    }
                    desc = newsfeed_pregnancy_mother
                }
                desc = newsfeed_pregnancy_father
            }
        }
        desc = {
            first_valid = {
                triggered_desc = {
                    trigger = { NOT = { exists = scope:father } }
                    desc = newsfeed_pregnancy_no_father_desc
                }
                desc = newsfeed_pregnancy_desc
            }
        }
    }
}
